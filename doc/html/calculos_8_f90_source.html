22 subroutine conversion
23 !$ use omp_lib
24 use vars_dat
25 implicit none
26 integer :: i,j,ii,jj,kl,l
27 integer :: ih
28 real*8 :: ylat1,ylat2,xlon1,xlon2

30 real*8 :: alat,alon,area,tot
31 real*8 :: xmas,xemis
32 
33 print *, "***** Doing interpolations ****"
34  do j= 2,djx-1
35  do i=2,dix-1
36  ylat1=.5*(dlat(i,j-1)+dlat(i,j))
37  ylat2=.5*(dlat(i,j+1)+dlat(i,j))
38  xlon1=.5*(dlon(i-1,j)+dlon(i,j))
39  xlon2=.5*(dlon(i+1,j)+dlon(i,j))
40  !$omp parallel do private(jj,area,ih,l,kl)
41  do ii=2,eix-1
42  do jj=2,ejx-1
43  alat=0.0
44  alon=0.0
45  elat1= 0.5*(elat(ii,jj)+elat(ii,jj-1))
46  elat2= 0.5*(elat(ii,jj)+elat(ii,jj+1))
47  elon1= 0.5*(elon(ii,jj)+elon(ii-1,jj))
48  elon2= 0.5*(elon(ii,jj)+elon(ii+1,jj))
49  tot=(elat2-elat1)*(elon2-elon1)/((ylat2-ylat1)*(xlon2-xlon1))
50  if(ylat1.le.elat2.and. ylat2.ge.elat1)alat=&
51  &(min(ylat2,elat2)-max(ylat1,elat1))/(elat2-elat1)
52  if(xlon1.le.elon2.and. xlon2.ge.elon1) alon=&
53  & (min(xlon2,elon2)-max(xlon1,elon1))/(elon2-elon1)
54  area=max(0.,alat*alon)* tot!
55  if( area.gt.0.) then
56  do l=1,size(ed,dim=3) ! compuesto
57  do ih=1,size(ed,dim=4) !hora
58  do kl=1,size(ed,dim=5) ! altura
59  if (tvar(kl)) ed(i,j,l,ih,kl)=ed(i,j,l,ih,kl)+ei(ii,jj,l,ih,kl)*area
60  end do ! kl
61  end do ! ih
62  end do ! l
63  if(tpob)dpob(i,j)=dpob(i,j)+epob(ii,jj)*area
64  end if
65  end do ! jj
66  end do ! ii
67  !$omp end parallel do
68  xmas=xmas+ed(i,j,1,1,8)*dx*dy/1000000
69  end do ! i
70 end do ! j
71  do i=1,eix
72  do j=1,ejx
73  xemis=xemis+ei(i,j,1,1,8) *dxe*dye/1000000
74  end do
75  end do
76 !
77 print *,'Mass balance'
78 print '(A20,F10.0,x,f7.0,x,f7.0)','Emissions Inventory:',xemis, dxe,dye
79 print '(A20,F10.0,x,f7.0,x,f7.0)','Emissions Domain:',xmas , dx,dy
80 print '(A20,x,f10.4,x,f10.4,"%")','Ratio EI/EDx100:', xemis/xmas,(xemis/xmas-1)*100
81 print '(A)','****** Done interpolation'
